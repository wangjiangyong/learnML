def performance_metric(y_true, y_predict):
    """"""
    y_true1 = np.array(y_true)
    y_predict1 = np.array(y_predict)
    ytmp = np.mean(y_true1)
    SStot = np.sum((y_true1 - ytmp)**2)
    SSreg = np.sum((y_predict1 - ytmp)**2)
    SSres = np.sum((y_true1 - y_predict1)**2)
    
    score = 1.0 - SSres/SStot

    return score

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    split_tmp = len(X)/cv
    split0_X_train = X[:split_tmp* 2]    
    split0_X_test = X[split_tmp* 2:]
    split0_y_train = y[:split_tmp* 2]
    split0_y_test = y[split_tmp* 2:]
    
    split1_X_train = X[split_tmp:]    
    split1_X_test = X[:split_tmp]
    split1_y_train = y[split_tmp:]
    split1_y_test = y[:split_tmp]
    
    split2_X_train = X[:split_tmp].append(X[split_tmp* 2:])    
    split2_X_test = X[split_tmp:split_tmp* 2]
    split2_y_train = y[:split_tmp].append(y[split_tmp* 2:])
    split2_y_test = y[split_tmp:split_tmp* 2]
    
    estimator0 = estimator.fit(split0_X_train,split0_y_train)
    split0_y_predict = estimator0.predict(split0_X_test)
    split0_score = scoring(split0_y_test,split0_y_predict)
    
    estimator1 = estimator.fit(split1_X_train,split1_y_train)
    split1_y_predict = estimator1.predict(split1_X_test)
    split1_score = scoring(split1_y_test,split1_y_predict)
    
    estimator2 = estimator.fit(split2_X_train,split2_y_train)
    split2_y_predict = estimator2.predict(split2_X_test)
    split2_score = scoring(split2_y_test,split2_y_predict)
    
    
    
    scores = [split0_score,split1_score,split2_score]
    return scores


def fit_model(X, y):
    """  [X,y]"""
    if len(X) != len(y):
        return None
    
    tmpBest = 0.0
    tmpBestregressor = None
    for i in range(10):
        regressor = DecisionTreeRegressor(max_depth=i+1)
        tmp_mean_score = np.mean(cross_val_score(regressor, X, y))
        if  tmp_mean_score > tmpBest:
            tmpBest = tmp_mean_score
            tmpBestregressor = regressor
        
    #
    best_estimator = tmpBestregressor
    
    return best_estimator
